	listen 443 ssl;
    	server_name slack.kinto-storage.org;
	
	ssl_certificate /etc/letsencrypt/live/slack.kinto-storage.org/fullchain.pem;
	ssl_certificate_key /etc/letsencrypt/live/slack.kinto-storage.org/privkey.pem;

	include /home/poils/dev/certbot/certbot-nginx/certbot_nginx/options-ssl-nginx.conf;


        location / {
          proxy_set_header        Host $host;
          proxy_set_header        X-Real-IP $remote_addr;
          proxy_set_header        X-Forwarded-For $proxy_add_x_forwarded_for;

          proxy_pass          http://127.0.0.1:9008;
          proxy_read_timeout  90;

          proxy_redirect      http://127.0.0.1:9008 https://slack.kinto-storage.org;
        }
